def partition(partition_list, left, right):
swap(partition_list, left, right)
pivot = partition_list[right]
wall = left
current_idx = left
while current_idx < right:
if partition_list[current_idx] < pivot:
swap(partition_list, current_idx, wall)
wall += 1
current_idx += 1
swap(partition_list, right, wall)
return wall

from math import floor
def binary_search(input_list, search, offset=0):
if len(input_list) == 1:
if input_list[0] == search:
return offset
else:
return -1
midpoint = int(floor(len(input_list)/2))
if search > input_list[midpoint]:
offset += midpoint
return binary_search(input_list[midpoint:], search, offset)
elif search < input_list[midpoint]:
return binary_search(input_list[:midpoint], search, offset)
else:
return midpoint + offset
